SELECT APELLIDOS,NOMBRES,SEXO,NACIONALIDAD,TIPO_DOC,NRO_DOC,
TO_CHAR(FECHA_CRUCE,'dd/mm/yyyy hh24:mi:ss') AS FECHA_CRUCE,
PASO_CRUCE,TIPO_CRUCE FROM (SELECT TRIM(APELLIDO1 || ' ' || APELLIDO2) AS APELLIDOS,TRIM(NOMBRE1 || ' ' || NOMBRE2) AS NOMBRES,SEXO,NACIONALIDAD,TIPO_DOC,NRO_DOC,FECHA_CRUCE,PASO_CRUCE,TIPO_CRUCE FROM V_WEB_SERVICE_OL WHERE NOMBRE1 LIKE 'JUAN%' AND NOMBRE2 LIKE 'PABLO%' AND APELLIDO1 LIKE 'BUDROVICH%' AND NACIONALIDAD = 'ARGENTINA' AND FECHA_NAC = to_date('05/08/1981','dd/mm/yyyy') AND ESTADO = 'ACEPTADO' ) WHERE ROWNUM <=501
